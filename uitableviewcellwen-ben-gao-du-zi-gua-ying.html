UITableViewCell文本高度自适应

Date 二 09 九月 2014 Tags iOS

在UITableViewCell中使用UILabel存储文本，每个cell的高度需要根据其文本内容的长度自适应。

Cell的实现

头文件:

//
//  SelfAdaptionTableViewCell.h
//

#import <UIKit/UIKit.h>

@interface SelfAdaptionTableViewCell : UITableViewCell
@property(nonatomic,strong)    UILabel *content;
//实现自动换行
-(void)setContentText:(NSString*)text;
@end

m文件:

//
//  SelfAdaptionTableViewCell.m
//
#import "SelfAdaptionTableViewCell.h"

@implementation SelfAdaptionTableViewCell

- (id)initWithStyle:(UITableViewCellStyle)style reuseIdentifier:(NSString *)reuseIdentifier
{
    self = [super initWithStyle:style reuseIdentifier:reuseIdentifier];
    if (self) {
        [self initLayout];
    }
    return self;
}

- (void)initLayout
{
    self.content=[[UILabel alloc]initWithFrame:CGRectMake(0, 0, self.width-30, 35)];
    self.content.textAlignment=UITextAlignmentLeft;
    self.content.font=[UIFont systemFontOfSize:13.0];
    [self.contentView addSubview:self.content];
}

- (void)setContentText:(NSString *)text
{
    //获得当前cell高度
    CGRect frame = [self frame];

    //文本赋值
    self.content.text = text;

    //设置label的最大行数
    self.content.numberOfLines = 1000;

    CGSize size = CGSizeMake(self.width-30, 10000);//高度设置大一些

    //计算出自适应的高度
    CGSize labelSize = [self.content.text sizeWithFont:self.content.font constrainedToSize:size lineBreakMode:NSLineBreakByClipping];

    self.content.frame = CGRectMake(self.content.frame.origin.x, self.content.frame.origin.y, labelSize.width, labelSize.height);
    frame.size.height = labelSize.height;
    self.frame = frame;
}
@end

Cell的使用

//
//  TestViewController.h
//
#import <UIKit/UIKit.h>

@interface TestViewController : UIViewController<UITableViewDataSource,UITableViewDelegate>
@property (nonatomic,strong) NSArray *contentArray;
@end



//
//  TestViewController.m
//
@interface TestViewController () {
    UITableView *_contentTV;
}
@end

@implementation TestViewController

- (id)initWithNibName:(NSString *)nibNameOrNil bundle:(NSBundle *)nibBundleOrNil
{
    self = [super initWithNibName:nibNameOrNil bundle:nibBundleOrNil];
    if (self) {
        self.title=@"自适应高度Cell测试";
    }
    return self;
}

- (void)viewDidLoad
{
    _contentTV=[[UITableView alloc]initWithFrame:CGRectMake(8, 0, self.view.width-8*2, self.view.height)  style:UITableViewStylePlain];
    _contentTV.delegate = self;
    _contentTV.dataSource = self;
    [self.view addSubview:_contentTV];
}

- (NSInteger)tableView:(UITableView *)tableView numberOfRowsInSection:(NSInteger)section
{
    return [self.contentArray count];
}

- (UITableViewCell *)tableView:(UITableView *)tableView cellForRowAtIndexPath:(NSIndexPath *)indexPath
{
    static NSString *CellIdentifier = @"selfAdaptionCell";

    SelfAdaptionTableViewCell *cell =  [tableView dequeueReusableCellWithIdentifier:CellIdentifier];
    if (cell == nil) {
        cell = [[SelfAdaptionTableViewCell alloc] initWithStyle:UITableViewCellStyleDefault reuseIdentifier:CellIdentifier];
    }
    [cell setIntroductionText: [self.contentArray objectAtIndex:indexPath.row];
    return cell;
}

-(NSInteger)numberOfSectionsInTableView:(UITableView *)tableView{
    return 1;
}
-(CGFloat)tableView:(UITableView *)tableView heightForRowAtIndexPath:(NSIndexPath *)indexPath
{
    SelfAdaptionTableViewCell *cell = [self tableView:_contentTV cellForRowAtIndexPath:indexPath];
    return cell.frame.size.height;
}
